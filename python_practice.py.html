#Problem statement -  Find remainder of array multiplication divided by n


# with static approach

def arr_mul_remainder(arr1, n):

    mul = 1
    for i in range(len(arr1)):
        mul *= arr1[i]

    return mul % n

arr1 = [100, 10, 5, 25, 35, 14]
n = 11

print(arr_mul_remainder(arr1, n))



#dynamic approach

class ArrayOperations:
    def __init__(self, arr):
        self.arr = arr

    def product(self):
        result = 1
        for num in self.arr:
            result *= num
        return result

    def remainder(self, n):
        return self.product() % n

# Get user input for the array
arr_str = input("Enter the elements of the array: ")
arr1 = [int(x) for x in arr_str.split()]

# Get user input for the value of n
n = int(input("Enter the value of n: "))

# Create an instance of ArrayOperations
arr_ops = ArrayOperations(arr1)

# Calculate the product
product_result = arr_ops.product()

# Print the product
print("Product of the elements in the array:", product_result)

# Calculate and print the remainder
print("Remainder of the product when divided by n:", product_result % n)
